class GreenZoneIndex:
    def __init__(self, territory_name, territory_area, green_zones):
        self.territory_name = territory_name
        self.territory_area = territory_area
        self.green_zones = green_zones
     

        self.green_index = self.calculate_green_index()  # индекс озеленения

    def calculate_green_index(self): 
        green_index = 0
        for current_green_zone in self.green_zones:           
            current_green_index = (current_green_zone / self.territory_area)*100 
            green_index = (green_index + current_green_index)

        return round(green_index, 2)

Pushkin = GreenZoneIndex("Пушкин", territory_area=28676, green_zones=[302, 487, 420, 325, 471, 363, 404])
print(Pushkin.green_index)
